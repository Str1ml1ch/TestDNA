LEFT JOIN Подкатегория_1 ON Подкатегория_1.id_Подкатегории = Подкатегория_0.id
LEFT JOIN Товар_Подкатегория_1 ON Товар_Подкатегория_1.id_Подкатегория_1 = Подкатегория_1.id
LEFT JOIN Товар ON Товар.id = Товар_Подкатегория_1.id_Товар
WHERE Товар.id IN (1,2,3)
